Docker 基本操作

2023/10/20 posted in  Linux
Tags: 

安装 Docker

下载 docker.sh 文件

curl -o docker.sh -ssL https://get.docker.com

# 执行权限
chmod +x docker.sh

以阿里云镜像安装 docker.sh 脚本内容

sudo sh docker.sh --mirror Aliyun
yum install -y -q docker-ce

启动 Docker

systemctl start docker

# 设置开机启动 Docker
systemctl enable docker

容器命令

# 使用此目录的 Dockerfile 创建镜像
docker build -t friendlyname

# 运行端口 4000 到 80 的“名称”映射
docker run -p 4000:80 friendlyname	

# 内容相同，但在分离模式下
docker run -d -p 4000:80 friendlyname

# 查看所有正在运行的容器的列表
docker ps

# 查看所有容器的列表，甚至包含未运行的容器
docker ps -a

# 平稳地停止指定的容器
docker stop <CONTAINER ID>	

# 强制关闭指定的容器
docker kill <CONTAINER ID>	

# 从此机器中删除指定的容器
docker rm <CONTAINER ID>	

#  从此机器中删除所有容器
docker rm $(docker ps -a -q)

镜像命令

# 显示此机器上的所有镜像
docker images -a

# 从此机器中删除指定的镜像
docker rmi <imagename>	

# 从此机器中删除所有镜像
docker rmi $(docker images -q)	

# 进入正在运行的镜像
docker exec -it mysql bash

# 退出正在运行的镜像
exit

Docker 仓库命令


# 使用您的 Docker 凭证登录此 CLI 会话
docker login

# 标记 <image> 以上传到镜像库
docker tag <image> username/repository:tag

# 将已标记的镜像上传到镜像库
docker push username/repository:tag

# 拉取
docker pull username/repository:tag

# 运行镜像库中的镜像
docker run -p 4000:80 username/repository:tag

# 后台运行镜像库中的镜像
docker run -d -p 4000:80 username/repository:tag

# 示例
docker run -d -p 10022:22 -p 10080:3000 --name=gogs -v /var/gogs/:/data penghuicai/gogs:0.11.53

# 查看信息
docker inspect gogs

# 提交修改后的镜像
docker commit CONTAINER_ID imagename
